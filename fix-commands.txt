# VPS da bajariladigan komandalar (ketma-ket)

# 1. Xatoni ko'rish
pm2 logs salary-backend --lines 50 --nostream

# 2. Eski processni o'chirish
pm2 delete salary-backend

# 3. Loyiha papkasiga o'tish (to'g'ri yo'lni aniqlash)
cd /root/projects
ls -la

# 4. Salary papkasini topish
find /root -name "salary" -type d 2>/dev/null | grep -v node_modules

# 5. To'g'ri papkaga o'tish (masalan)
cd /root/projects/oylik-aliboboqurilish/salary

# 6. .env faylini tekshirish
cat .env

# 7. Dependencies o'rnatish
npm install

# 8. Backend ni ishga tushirish
pm2 start server/index.js --name "salary-backend" --time

# 9. Loglarni kuzatish
pm2 logs salary-backend

# 10. Agar muammo bo'lsa, qo'lda ishga tushirish
node server/index.js
